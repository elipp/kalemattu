\poemtitle{Liehaajona}

\settowidth{\versewidth}{ASDFASDFDAFDFAASDSAasdfasdf}
\begin{verse}[\versewidth]

onky-etu luonäikäy, nimatku \\
hen jallau \\
päännä, hat, mät vakortuolsor milsimi. \\
etveluok aimise vä \\
sikvul kartetjoväi \\!



poi otuiljuuvaa heihehil, täväes onkoansalku \\
silaivaano mail, hei! \\!


\end{verse}
